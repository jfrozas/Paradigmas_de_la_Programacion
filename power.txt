let rec power x y = 
	if (y>0)then x * power x (y-1) else 1;;

let rec power' x y = 
  if y =0 then 1 else
  if (y mod 2 = 0) then power' (x*x) (y/2) else x * power' (x*x) (y/2);;
 
 (*power' es m√°s eficiente porque diferencia entre exponentes pares y impares pero al realizarse en int no se llega a apreciar una gran diferencia  *)
 
let rec powerf x y = 
  if y = 0 then 1. else
  if (y mod 2 = 0) then powerf (x*.x) (y/2) else x *. powerf (x*.x) (y/2);;
